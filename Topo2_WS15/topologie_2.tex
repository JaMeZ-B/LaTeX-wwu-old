\section{Vektorbündel} % (fold)
\label{sec:2}
Wir betrachten nur reelle und komplexe Vektorbündel, es sei also $K \in \set{\mathds{R},\mathds{C}}$.

\begin{definition}[{name=[$K$-Vektorbündel]},label=def_vektbuendel]
Sei $X$ ein topologischer Raum. Eine stetige surjektive Abbildung $p \colon E \to X$ heißt $K$-\Index{Vektorbündel}, falls gilt: 
\begin{enumerate}[a)]
	\item $ E_x := p^{-1}(\set{x})$ ist ein $K$-Vektorraum für alle $x \in X$. $E_x$ heißt die \Index{Faser} von $E$ über $x$. 
	\item Für alle Punkte $x \in X$ gibt es eine Umgebung $U$, ein $n \in \mathds{N}$ und einen Homöomorphismus $\varphi_U \colon p^{-1}(U) \to U \times K^n$, sodass das 
	folgende 
	Diagramm kommutiert
	\[
		\begin{tikzcd}
			p^{-1}(U) \arrow["\varphi_U"]{rr} \drar["p"'] & & U \times K^n  \dlar["\mathrm{pr}_U"] \\
			& U 
		\end{tikzcd}
	\]
	und die Einschränkung $\varphi_U |_{E_x} \colon E_x \to K^n$ für alle $x \in U$ ein Isomorphismus von $K$"=Vektorräumen ist. Diese Eigenschaft wird oft als 
	\Index{lokale Trivialität} bezeichnet.
\end{enumerate}
\end{definition}
