\chapter{CAT(0)-Räume} % (fold)
\label{cha:1}

\begin{definition}[metrischer Raum]
	Sei $X$ eine Menge. Eine Abbildung $d \colon X \times X \rightarrow \RR_{\geq 0}$ heißt \Index{Metrik}, wenn für alle $x,y,z \in X$ gilt:
	\begin{enumerate}[(i)]
		\item $d(x,y) = 0 \Leftrightarrow x=y$
		\item $d(x,y) = d(y,x)$
		\item $d(x,z) \leq d(x,y) + d(y,z)$
	\end{enumerate}
	Das Paar $(X,d)$ heißt dann \Index{metrischer Raum}.
\end{definition}
